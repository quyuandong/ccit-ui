使用 ​

全局注册 ​

ts
// main.ts
// 引入组件
import JyUI from '@junyang/jy-ui'
// 引入样式
import '@junyang/jy-ui/lib/style.css'

// 使用组件
createApp(App).use(JyUI).mount('#app')
// main.ts
// 引入组件
import JyUI from '@junyang/jy-ui'
